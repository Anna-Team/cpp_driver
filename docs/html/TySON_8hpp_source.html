99 void parse_map_elements(std::string_view object)
100 {
101 auto end_type_sep = object.find_first_of('{') + 1;
102 auto end_value_sep = (object.size() - 1) - end_type_sep;
103 auto map_data = object.substr(end_type_sep, end_value_sep);
104
105 auto vec_data = utils::split(map_data, ',');
106
107 auto to_map_element = [this](auto &val) -> std::vector<TySonObject>
108 {
109 return parse_map_element(val);
110 };
111
112 std::vector<std::vector<TySonObject>> result;
113 std::transform(vec_data.begin(), vec_data.end(), std::back_inserter(result), to_map_element);
114 std::for_each(result.begin(), result.end(), [this](auto key_val)
115 {
116 map_.try_emplace(key_val[0], key_val[1]);
117 });
118 }

121 {

136 case TySonType::Vector:

138 std::stringstream sstream;
139 std::for_each(obj.vector_.begin(),
140 obj.vector_.end(),
141 [&sstream](const auto &val){ sstream << val << ","; });

143 return out << "v[" << sstream.str() << "]";
144 }
145 case TySonType::Map:
146 {
147 std::stringstream sstream;
148 std::for_each(obj.map_.begin(),
149 obj.map_.end(),
150 [&sstream](const std::pair<TySonObject, TySonObject> &val)
151 {
152 sstream << val.first << ":" << val.second << ",";
153 });
154
155 return out << "m{" << sstream.str() << "}";
156 }

173 explicit TySonObject(std::string_view object)
174 {
175 auto end_type_sep = object.find_first_of('|');
176 auto end_value_sep = (object.size() - 1) - (end_type_sep + 1);
177 auto type = object.substr(0, end_type_sep);
178
179 if (object == "null")
180 {
181 type_ = TySonType::Null;
182 value_ = {};
183 return;
184 }
185
186 if (object.starts_with("uts"))
187 {
188 type_ = TySonType::Timestamp;
189 value_ = object.substr(end_type_sep + 1, end_value_sep);
190 }
191 else if (object.starts_with('v'))
192 {
193 type_ = TySonType::Vector;
194
195 parse_vector_elements(object);
196 }
197 else if (object.starts_with('m'))
198 {
199 type_ = TySonType::Map;
200 parse_map_elements(object);
201 }
202 else if (type.size() > 1)
203 {
204 type_ = TySonType::Link;
205 auto val = object.substr(end_type_sep + 1, end_value_sep);
206 link_ = std::make_pair(type, val);
207 }
208 else
209 {
210 type_ = static_cast<TySonType>(type.at(0));
211 value_ = object.substr(end_type_sep + 1, end_value_sep);
212 }
213 }

467 template<typename T>
468 requires std::is_arithmetic_v<T>
469 [[nodiscard]] T value() const
470 {
471 if (type_ == TySonType::Number || type_ == TySonType::Bool || type_ == TySonType::Timestamp)
472 {
473 std::size_t pos{};
474
475 switch (*typeid(T).name())
476 {
477 case 's':
478 case 'j':
479 case 'i':
480 return std::stoi(value_, &pos);
481 case 'l':
482 return std::stol(value_, &pos);
483 case 'f':
484 return static_cast<T>(std::stof(value_, &pos));
485 case 'd':
486 return static_cast<T>(std::stod(value_, &pos));
487 case 'x':
488 return std::stoll(value_, &pos);
489 case 'e':
490 return static_cast<T>(std::stold(value_, &pos));
491 case 'b':
492 return value_ == "true";
493 case 'c':
494 return value_.c_str()[0];
495 }
496 }
497
498 throw std::invalid_argument("Invalid Type");
499 }
