20 KeyVal(std::string data)
21 {
22 auto separation = data.find_first_of(':');

103 template<tyson::TySonType T>
104 requires (T == tyson::TySonType::Objects || T == tyson::TySonType::IDs)
105 std::optional<tyson::TySonCollectionObject> get()
106 {
107 if (data_.starts_with(std::string("s|data|:objects")) && T == tyson::TySonType::Objects)
108 {
109 tyson::TySonCollectionObject object {};
110 auto start_val = data_.find_first_of('{') + 1;
111 auto end_val = data_.find_last_of('}');
112
113 auto tyson_str_data = split_data(data_.substr(start_val, end_val - start_val));
114 for (auto &key_val: tyson_str_data)
115 {
116 object.add(std::make_pair(key_val.link, key_val.value));
117 }
118
119 tyson_str_data.clear();
120 tyson_str_data.shrink_to_fit();
121
122 return object;
123 }
124 else if (data_.starts_with(std::string("s|data|:ids")) && T == tyson::TySonType::IDs)
125 {
126 tyson::TySonCollectionObject object {};
127 auto start_val = data_.find_first_of('[');
128 auto end_val = data_.find_last_of(']');
129
130 auto tyson_str_data = utils::split(data_.substr(start_val, end_val - start_val), ',');
131 for (auto &link_data: tyson_str_data)
132 {
133 object.add(link_data);
134 }
135
136 tyson_str_data.clear();
137 tyson_str_data.shrink_to_fit();
138
139 return object;
140 }

247 void parse_response(std::string_view response)
248 {
249 /*
250 * The response format from annadb is
251 * `result:<ok|false>[response{s|data|<...>, s|meta|<...>}]`
252 */
253 auto pos_result_end = response.find('[');
254 auto pos_data_begin = response.find("s|data|:");
255
256 // the meta information is always at the end,
257 // so that it is faster to use rfind instead of regular find
258 auto pos_meta_begin = response.rfind("s|meta|:");
259
260 // there is no usage for the closing tags, so we exclude them
261 auto pos_response_end = response.rfind(",}]");
262
263 auto res = response.substr(0, pos_result_end);
264 if (res.find("ok") != std::string::npos)
265 {
266 result_ = true;
267 }
268
269 data_ = response.substr(pos_data_begin, pos_meta_begin - pos_data_begin);
270 meta_ = response.substr(pos_meta_begin, pos_response_end - pos_meta_begin);
271 }
272
273 friend std::ostream & operator<<(std::ostream &os, const Journal& journal)
274 {
275 std::string result = journal.ok() ? "ok" : "err";
276 auto meta = journal.meta();
277 auto data = "journal.data()"; // journal.data();
278
279 std::string repr = "result:" + result + "[response{";
280 return os << repr << data << meta << ",},];";
281 }
