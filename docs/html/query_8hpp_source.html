574 class Query
575 {
576 std::string collection_name_;
577 std::vector<std::unique_ptr<annadb::Query::QueryCmd>> cmds_ {};
578 void add_to_cmds(std::unique_ptr<QueryCmd> queryCmd)
579 {
580 if (cmds_.empty() && !queryCmd->can_start_pipeline())
581 {
582 throw std::invalid_argument(queryCmd->name() + " can not be used to start a new query pipeline.");
583 }
584
585 if (cmds_.empty())
586 {
587 this->cmds_.emplace_back(std::move(queryCmd));
588 }
589 else
590 {
591 auto latest_cmd = cmds_.size() - 1;
592 if (cmds_.at(latest_cmd)->next_step_allowed(queryCmd->name()) &&
593 queryCmd->previous_step_allowed(cmds_.at(latest_cmd)->name()))
594 {
595 this->cmds_.emplace_back(std::move(queryCmd));
596 }
597 else
598 {
599 throw std::invalid_argument(queryCmd->name() + " can not be used before/after " + cmds_.at(latest_cmd)->name());
600 }
601 }
602 }
603
604 friend std::ostream& operator<<(std::ostream &out, Query &query)
605 {
606 std::stringstream sstream;
607 sstream << "collection|" << query.collection_name_ << "|";
608
609 if (query.cmds_.size() == 1)
610 {
611 sstream << ":" << query.cmds_[0]->query() << ";";
612 }
613 else
614 {
615 sstream << ":q[";
616 for (auto &cmd: query.cmds_)
617 {
618 sstream << cmd->query() << ",";
619 }
620 sstream << "];";
621 }
622 return out << sstream.str();
623 }
