86 virtual ~QueryCmd() = default;
87
88 [[nodiscard]] std::string name()
89 {
90 return this->name_;
91 }
92
93 [[nodiscard]] bool can_start_pipeline() const
94 {
95 return start_cmd_;

97

99 {
100 return this->annadb_query();
101 }
102
103 [[nodiscard]] bool next_step_allowed(const std::string &cmdName)
104 {
105 if (next_steps_().empty())
106 {
107 return false;
108 }
109
110 auto res = std::find(next_steps_().begin(), next_steps_().end(), cmdName);
111 return res != next_steps_().end();
112 }
113
114 [[nodiscard]] bool previous_step_allowed(const std::string &cmdName)
115 {
116 if (previous_steps_().empty())
117 {
118 return false;
119 }
120
121 auto res = std::find(previous_steps_().begin(), previous_steps_().end(), cmdName);
122 return res != next_steps_().end();
123 }

739 class Query
740 {
741 std::string collection_name_;
742 std::vector<std::unique_ptr<annadb::Query::QueryCmd>> cmds_ {};
743 void add_to_cmds(std::unique_ptr<QueryCmd> queryCmd)
744 {
745 if (cmds_.empty() && !queryCmd->can_start_pipeline())
746 {
747 throw std::invalid_argument(queryCmd->name() + " can not be used to start a new query pipeline.");
748 }
749
750 if (cmds_.empty())
751 {
752 this->cmds_.emplace_back(std::move(queryCmd));
753 }
754 else
755 {
756 auto latest_cmd = cmds_.size() - 1;
757 if (cmds_.at(latest_cmd)->next_step_allowed(queryCmd->name()) &&
758 queryCmd->previous_step_allowed(cmds_.at(latest_cmd)->name()))
759 {
760 this->cmds_.emplace_back(std::move(queryCmd));
761 }
762 else
763 {
764 throw std::invalid_argument(queryCmd->name() + " can not be used before/after " + cmds_.at(latest_cmd)->name());
765 }
766 }
767 }
768
769 friend std::ostream& operator<<(std::ostream &out, Query &query)
770 {
771 std::stringstream sstream;
772 sstream << "collection|" << query.collection_name_ << "|";
773
774 if (query.cmds_.size() == 1)
775 {
776 sstream << ":" << query.cmds_[0]->query() << ";";
777 }
778 else
779 {
780 sstream << ":q[";
781 for (auto &cmd: query.cmds_)
782 {
783 sstream << cmd->query() << ",";
784 }
785 sstream << "];";
786 }
787 return out << sstream.str();
788 }

1005 void update(UpdateType &kind, T&& ...values)
1006 {
